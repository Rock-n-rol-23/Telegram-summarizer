```python
import aiofiles
import os
import tempfile
from telegram import Update
from telegram.ext import ContextTypes

async def download_telegram_file(update: Update, context: ContextTypes.DEFAULT_TYPE):
    """–°–∫–∞—á–∏–≤–∞–µ—Ç —Ñ–∞–π–ª, –æ—Ç–ø—Ä–∞–≤–ª–µ–Ω–Ω—ã–π –ø–æ–ª—å–∑–æ–≤–∞—Ç–µ–ª–µ–º –≤ Telegram"""
    try:
        # –ü–æ–ª—É—á–∞–µ–º –∏–Ω—Ñ–æ—Ä–º–∞—Ü–∏—é –æ —Ñ–∞–π–ª–µ
        if update.message.document:
            file_info = await context.bot.get_file(update.message.document.file_id)
            file_name = update.message.document.file_name
            file_size = update.message.document.file_size
        else:
            return {
                'success': False,
                'error': '–§–∞–π–ª –Ω–µ –Ω–∞–π–¥–µ–Ω –≤ —Å–æ–æ–±—â–µ–Ω–∏–∏'
            }
        
        # –ü—Ä–æ–≤–µ—Ä—è–µ–º —Ä–∞–∑–º–µ—Ä —Ñ–∞–π–ª–∞ (–º–∞–∫—Å–∏–º—É–º 20MB –æ—Ç Telegram)
        if file_size > 20 * 1024 * 1024:
            return {
                'success': False,
                'error': '–§–∞–π–ª —Å–ª–∏—à–∫–æ–º –±–æ–ª—å—à–æ–π (–º–∞–∫—Å–∏–º—É–º 20MB)'
            }

        allowed_extensions = ['.pdf', '.docx', '.doc', '.txt']
        file_extension = os.path.splitext(file_name.lower())[1]
        
        if file_extension not in allowed_extensions:
            return {
                'success': False,
                'error': f'–ù–µ–ø–æ–¥–¥–µ—Ä–∂–∏–≤–∞–µ–º—ã–π —Ñ–æ—Ä–º–∞—Ç —Ñ–∞–π–ª–∞. –ü–æ–¥–¥–µ—Ä–∂–∏–≤–∞—é—Ç—Å—è: {", ".join(allowed_extensions)}'

        temp_dir = tempfile.mkdtemp()
        local_file_path = os.path.join(temp_dir, file_name)
        
        # –°–∫–∞—á–∏–≤–∞–µ–º —Ñ–∞–π–ª

            'success': True,
            'file_path': local_file_path,
            'file_name': file_name,
            'file_size': file_size,
            'file_extension': file_extension,
            'temp_dir': temp_dir
        }
        
    except Exception as e:
        return {
            'success': False,
            'error': f'–û—à–∏–±–∫–∞ –ø—Ä–∏ —Å–∫–∞—á–∏–≤–∞–Ω–∏–∏ —Ñ–∞–π–ª–∞: {str(e)}'
        }
```

```python
import PyPDF2
import pdfplumber

def extract_text_from_pdf(file_path):
    """–ò–∑–≤–ª–µ–∫–∞–µ—Ç —Ç–µ–∫—Å—Ç –∏–∑ PDF —Ñ–∞–π–ª–∞"""
    try:
        text = ""
        
        # –°–Ω–∞—á–∞–ª–∞ –ø—Ä–æ–±—É–µ–º pdfplumber (–ª—É—á—à–µ –¥–ª—è —Å–ª–æ–∂–Ω—ã—Ö PDF)
        try:
            with pdfplumber.open(file_path) as pdf:
                for page in pdf.pages:
                    page_text = page.extract_text()
                    if page_text:
                        text += page_text + "\n\n"
                        
            if text.strip():
                return {
                    'success': True,
                    'text': text.strip(),
                    'method': 'pdfplumber'
                }
        except Exception as e:
            print(f"pdfplumber –Ω–µ —Å—Ä–∞–±–æ—Ç–∞–ª: {e}")
        
        # Fallback –Ω–∞ PyPDF2
        try:
            with open(file_path, 'rb') as file:
                reader = PyPDF2.PdfReader(file)
                
                # –ü—Ä–æ–≤–µ—Ä—è–µ–º, –∑–∞—à–∏—Ñ—Ä–æ–≤–∞–Ω –ª–∏ PDF
                if reader.is_encrypted:
                    return {
                        'success': False,
                        'error': 'PDF —Ñ–∞–π–ª –∑–∞—â–∏—â–µ–Ω –ø–∞—Ä–æ–ª–µ–º'
                    }
                
                for page in reader.pages:
                    page_text = page.extract_text()
                    if page_text:
                        text += page_text + "\n\n"
                        
            if text.strip():
                return {
                    'success': True,
                    'text': text.strip(),
                    'method': 'PyPDF2'
                }
            else:
                return {
                    'success': False,
                    'error': 'PDF –Ω–µ —Å–æ–¥–µ—Ä–∂–∏—Ç –∏–∑–≤–ª–µ–∫–∞–µ–º–æ–≥–æ —Ç–µ–∫—Å—Ç–∞ (–≤–æ–∑–º–æ–∂–Ω–æ, —Ç–æ–ª—å–∫–æ –∏–∑–æ–±—Ä–∞–∂–µ–Ω–∏—è)'
                }
                
        except Exception as e:
            return {
                'success': False,
                'error': f'–û—à–∏–±–∫–∞ —á—Ç–µ–Ω–∏—è PDF: {str(e)}'
            }
            
    except Exception as e:
        return {
            'success': False,
            'error': f'–û—à–∏–±–∫–∞ –æ–±—Ä–∞–±–æ—Ç–∫–∏ PDF: {str(e)}'
        }
```

### –§—É–Ω–∫—Ü–∏—è 3: –ò–∑–≤–ª–µ—á–µ–Ω–∏–µ —Ç–µ–∫—Å—Ç–∞ –∏–∑ DOCX

```python
from docx import Document

def extract_text_from_docx(file_path):
    """–ò–∑–≤–ª–µ–∫–∞–µ—Ç —Ç–µ–∫—Å—Ç –∏–∑ DOCX —Ñ–∞–π–ª–∞"""
    try:
        doc = Document(file_path)
        text = ""
        
        # –ò–∑–≤–ª–µ–∫–∞–µ–º —Ç–µ–∫—Å—Ç –∏–∑ –ø–∞—Ä–∞–≥—Ä–∞—Ñ–æ–≤
        for paragraph in doc.paragraphs:
            if paragraph.text.strip():
                text += paragraph.text + "\n"
        
        # –ò–∑–≤–ª–µ–∫–∞–µ–º —Ç–µ–∫—Å—Ç –∏–∑ —Ç–∞–±–ª–∏—Ü
        for table in doc.tables:
            for row in table.rows:
                for cell in row.cells:
                    if cell.text.strip():
                        text += cell.text + " "
                text += "\n"
        
        if text.strip():
            return {
                'success': True,
                'text': text.strip(),
                'method': 'python-docx'
            }
        else:
            return {
                'success': False,
                'error': 'DOCX —Ñ–∞–π–ª –Ω–µ —Å–æ–¥–µ—Ä–∂–∏—Ç —Ç–µ–∫—Å—Ç–∞'
            }
            
    except Exception as e:
        return {
            'success': False,
            'error': f'–û—à–∏–±–∫–∞ —á—Ç–µ–Ω–∏—è DOCX: {str(e)}'
        }
```

### –§—É–Ω–∫—Ü–∏—è 5: –ò–∑–≤–ª–µ—á–µ–Ω–∏–µ —Ç–µ–∫—Å—Ç–∞ –∏–∑ TXT

```python
import chardet

def extract_text_from_txt(file_path):
    """–ò–∑–≤–ª–µ–∫–∞–µ—Ç —Ç–µ–∫—Å—Ç –∏–∑ TXT —Ñ–∞–π–ª–∞ —Å –∞–≤—Ç–æ–æ–ø—Ä–µ–¥–µ–ª–µ–Ω–∏–µ–º –∫–æ–¥–∏—Ä–æ–≤–∫–∏"""
    try:
        # –û–ø—Ä–µ–¥–µ–ª—è–µ–º –∫–æ–¥–∏—Ä–æ–≤–∫—É —Ñ–∞–π–ª–∞
        with open(file_path, 'rb') as file:
            raw_data = file.read()
            encoding_result = chardet.detect(raw_data)
            encoding = encoding_result['encoding']
            
        if not encoding:
            encoding = 'utf-8'  # Fallback
        
        # –ß–∏—Ç–∞–µ–º —Ñ–∞–π–ª —Å –æ–ø—Ä–µ–¥–µ–ª–µ–Ω–Ω–æ–π –∫–æ–¥–∏—Ä–æ–≤–∫–æ–π
        try:
            with open(file_path, 'r', encoding=encoding) as file:
                text = file.read()
                
            return {
                'success': True,
                'text': text.strip(),
                'method': f'text file ({encoding})'
            }
            
        except UnicodeDecodeError:
            # –ü—Ä–æ–±—É–µ–º –¥—Ä—É–≥–∏–µ –ø–æ–ø—É–ª—è—Ä–Ω—ã–µ –∫–æ–¥–∏—Ä–æ–≤–∫–∏
            encodings_to_try = ['utf-8', 'cp1251', 'cp866', 'iso-8859-1', 'latin-1']
            
            for enc in encodings_to_try:
                try:
                    with open(file_path, 'r', encoding=enc) as file:
                        text = file.read()
                        
                    return {
                        'success': True,
                        'text': text.strip(),
                        'method': f'text file ({enc})'
                    }
                except:
                    continue
            
            return {
                'success': False,
                'error': '–ù–µ —É–¥–∞–ª–æ—Å—å –æ–ø—Ä–µ–¥–µ–ª–∏—Ç—å –∫–æ–¥–∏—Ä–æ–≤–∫—É —Ç–µ–∫—Å—Ç–æ–≤–æ–≥–æ —Ñ–∞–π–ª–∞'
            }
            
    except Exception as e:
        return {
            'success': False,
            'error': f'–û—à–∏–±–∫–∞ —á—Ç–µ–Ω–∏—è TXT —Ñ–∞–π–ª–∞: {str(e)}'
        }
```

```python
                },
                {
                    "role": "user", 
                    "content": prompt
                }
            ],
            model="llama3-8b-8192",
            max_tokens=max_tokens,
            temperature=0.3
        )
        
        summary = completion.choices[0].message.content.strip()
        
        return {
            'success': True,
            'summary': summary,
            'original_length': original_length,
            'processed_length': len(text)
        }
        
    except Exception as e:
        return {
            'success': False,
            'error': f'–û—à–∏–±–∫–∞ —Å—É–º–º–∞—Ä–∏–∑–∞—Ü–∏–∏ Groq: {str(e)}'
        }

def create_simple_document_summary(text, file_name=""):
    """–ü—Ä–æ—Å—Ç–æ–µ —Ä–µ–∑—é–º–µ –¥–æ–∫—É–º–µ–Ω—Ç–∞ –±–µ–∑ AI"""
    try:
        # –†–∞–∑–±–∏–≤–∞–µ–º –Ω–∞ –ø—Ä–µ–¥–ª–æ–∂–µ–Ω–∏—è
        sentences = text.replace('\n', ' ').split('.')
        sentences = [s.strip() for s in sentences if len(s.strip()) > 20]
        
        # –ë–µ—Ä–µ–º –∫–ª—é—á–µ–≤—ã–µ –ø—Ä–µ–¥–ª–æ–∂–µ–Ω–∏—è
        if len(sentences) <= 3:
            summary_sentences = sentences
        else:
            # –ë–µ—Ä–µ–º –ø–µ—Ä–≤–æ–µ, –Ω–µ—Å–∫–æ–ª—å–∫–æ —Å—Ä–µ–¥–Ω–∏—Ö –∏ –ø–æ—Å–ª–µ–¥–Ω–µ–µ –ø—Ä–µ–¥–ª–æ–∂–µ–Ω–∏–µ
            summary_sentences = [
                sentences[0],  # –ù–∞—á–∞–ª–æ
                sentences[len(sentences)//4],  # –ß–µ—Ç–≤–µ—Ä—Ç—å
                sentences[len(sentences)//2],  # –°–µ—Ä–µ–¥–∏–Ω–∞
                sentences[3*len(sentences)//4],  # –¢—Ä–∏ —á–µ—Ç–≤–µ—Ä—Ç–∏
                sentences[-1]  # –ö–æ–Ω–µ—Ü
            ]
        
        summary = '. '.join(summary_sentences)
        
        if len(summary) > 1500:
            summary = summary[:1500] + "..."
        
        summary += "\n\n‚ö†Ô∏è –ê–≤—Ç–æ–º–∞—Ç–∏—á–µ—Å–∫–æ–µ –∏–∑–≤–ª–µ—á–µ–Ω–∏–µ –∫–ª—é—á–µ–≤—ã—Ö –ø—Ä–µ–¥–ª–æ–∂–µ–Ω–∏–π (AI —Å—É–º–º–∞—Ä–∏–∑–∞—Ü–∏—è –Ω–µ–¥–æ—Å—Ç—É–ø–Ω–∞)"
        
        return {
            'success': True,
            'summary': summary,
            'original_length': len(text)
        }
        
    except Exception as e:
        return {
            'success': False,
            'error': f'–û—à–∏–±–∫–∞ —Å–æ–∑–¥–∞–Ω–∏—è –ø—Ä–æ—Å—Ç–æ–≥–æ —Ä–µ–∑—é–º–µ: {e}'
        }
```

```python
def split_large_text(text, max_chunk_size=12000):
    """–†–∞–∑–±–∏–≤–∞–µ—Ç –±–æ–ª—å—à–æ–π —Ç–µ–∫—Å—Ç –Ω–∞ —á–∞—Å—Ç–∏ –¥–ª—è –æ–±—Ä–∞–±–æ—Ç–∫–∏"""
    if len(text) <= max_chunk_size:
        return [text]
    
    # –†–∞–∑–±–∏–≤–∞–µ–º –ø–æ –∞–±–∑–∞—Ü–∞–º
    paragraphs = text.split('\n\n')
    chunks = []
    current_chunk = ""
    
    for paragraph in paragraphs:
        if len(current_chunk) + len(paragraph) + 2 <= max_chunk_size:
            current_chunk += paragraph + "\n\n"
        else:
            if current_chunk:
                chunks.append(current_chunk.strip())
            current_chunk = paragraph + "\n\n"
    
    if current_chunk:
        chunks.append(current_chunk.strip())
    
    return chunks
```
